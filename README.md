/* ======================= /src/screens/auth/LoginScreen.js ======================= */ import React, { useState } from 'react'; import { View, Text, TextInput, TouchableOpacity, ActivityIndicator } from 'react-native'; import DKLogo from '../../components/DKLogo'; import styles from '../../styles';

export default function LoginScreen({ navigation }) { const [email, setEmail] = useState(''); const [password, setPassword] = useState(''); const [loading, setLoading] = useState(false); async function handleLogin() { // implement firebase auth here if (!global.firebase) return alert('Install firebase and add config'); setLoading(true); try { await global.firebase.auth().signInWithEmailAndPassword(email, password); navigation.replace('HomeTabs'); } catch (e) { alert(e.message); } setLoading(false); } return ( <View style={styles.center}> <DKLogo size={80} /> <Text style={styles.title}>Welcome back to DK</Text> <TextInput placeholder="Email" value={email} onChangeText={setEmail} style={styles.input} /> <TextInput placeholder="Password" secureTextEntry value={password} onChangeText={setPassword} style={styles.input} /> <TouchableOpacity style={styles.button} onPress={handleLogin} disabled={loading}>{loading ? <ActivityIndicator color="#fff" /> : <Text style={styles.buttonText}>Log in</Text>}</TouchableOpacity> <TouchableOpacity onPress={() => navigation.navigate('Signup')}><Text style={styles.link}>Create an account</Text></TouchableOpacity> </View> ); }

/* ======================= /src/screens/auth/SignupScreen.js ======================= */ import React, { useState } from 'react'; import { View, Text, TextInput, TouchableOpacity, ActivityIndicator } from 'react-native'; import DKLogo from '../../components/DKLogo'; import styles from '../../styles'; export default function SignupScreen({ navigation }) { const [email, setEmail] = useState(''); const [password, setPassword] = useState(''); const [loading, setLoading] = useState(false); async function handleSignup() { if (!global.firebase) return alert('Install firebase and add config'); setLoading(true); try { await global.firebase.auth().createUserWithEmailAndPassword(email, password); navigation.replace('HomeTabs'); } catch (e) { alert(e.message); } setLoading(false); } return ( <View style={styles.center}> <DKLogo size={96} /> <Text style={styles.title}>Create your DK account</Text> <TextInput placeholder="Email" value={email} onChangeText={setEmail} style={styles.input} /> <TextInput placeholder="Password" secureTextEntry value={password} onChangeText={setPassword} style={styles.input} /> <TouchableOpacity style={styles.button} onPress={handleSignup} disabled={loading}>{loading ? <ActivityIndicator color="#fff" /> : <Text style={styles.buttonText}>Sign up</Text>}</TouchableOpacity> <TouchableOpacity onPress={() => navigation.navigate('Login')}><Text style={styles.link}>Already have an account? Log in</Text></TouchableOpacity> </View> ); }

/* ======================= /src/screens/ChatsScreen.js ======================= */ import React, { useEffect, useState } from 'react'; import { View, Text, FlatList, ActivityIndicator } from 'react-native'; import styles from '../styles'; export default function ChatsScreen() { const [messages, setMessages] = useState([]); const [loading, setLoading] = useState(true); useEffect(() => { if (!global.firebase) return setLoading(false); const db = global.firebase.firestore(); const unsub = db.collection('chats').orderBy('createdAt', 'desc').limit(50).onSnapshot(snap => { setMessages(snap.docs.map(d => ({ id: d.id, ...d.data() }))); setLoading(false); }, () => setLoading(false)); return () => unsub(); }, []); if (loading) return <View style={styles.center}><ActivityIndicator /></View>; return ( <View style={styles.screen}> <Text style={styles.heading}>Messages</Text> <FlatList data={messages} keyExtractor={i => i.id} renderItem={({ item }) => (<View style={styles.chatItem}><Text style={{ fontWeight: '700' }}>{item.senderName || 'Someone'}</Text><Text>{item.text}</Text></View>)} /> </View> ); }

/* ======================= /src/screens/ProfileScreen.js ======================= */ import React, { useState, useEffect } from 'react'; import { View, Text, TextInput, TouchableOpacity, Switch } from 'react-native'; import styles from '../styles'; import DKLogo from '../components/DKLogo'; import { useTheme } from '../context/ThemeContext'; export default function ProfileScreen({ navigation }) { const [profile, setProfile] = useState({ name: '', bio: '' }); const [loading, setLoading] = useState(false); const theme = useTheme(); useEffect(() => { if (!global.firebase) return; const user = global.firebase.auth().currentUser; if (!user) return; global.firebase.firestore().collection('profiles').doc(user.uid).get().then(d => { if (d.exists) setProfile(d.data()); }); }, []); async function saveProfile() { if (!global.firebase) return alert('Install firebase and add config'); setLoading(true); const user = global.firebase.auth().currentUser; try { await global.firebase.firestore().collection('profiles').doc(user.uid).set(profile, { merge: true }); alert('Saved'); } catch (e) { alert(e.message); } setLoading(false); } async function handleLogout() { if (!global.firebase) return navigation.replace('Login'); await global.firebase.auth().signOut(); navigation.replace('Login'); } return ( <View style={styles.screen}> <Text style={styles.heading}>Your profile</Text> <DKLogo size={88} /> <TextInput placeholder="Name" value={profile.name} onChangeText={(t) => setProfile(p => ({ ...p, name: t }))} style={styles.input} /> <TextInput placeholder="Bio" value={profile.bio} onChangeText={(t) => setProfile(p => ({ ...p, bio: t }))} style={styles.input} /> <TouchableOpacity style={styles.button} onPress={saveProfile} disabled={loading}><Text style={styles.buttonText}>{loading ? 'Saving...' : 'Save'}</Text></TouchableOpacity> <TouchableOpacity style={[styles.button, { backgroundColor: '#ccc' }]} onPress={handleLogout}><Text>Log out</Text></TouchableOpacity> <View style={{ flexDirection: 'row', alignItems: 'center', marginTop: 20 }}> <Text style={{ marginRight: 10 }}>Dark mode</Text> <Switch value={theme.dark} onValueChange={theme.toggle} /> </View> </View> ); }
